# calxo-robots-v1
# Default: allow all crawlers full access. We want both classic search engines
# and AI agents (ChatGPT, Claude, Perplexity, Apple Intelligence, etc.) to read
# Calxo's calculator pages so the formulas show up in answers.

User-agent: *
Allow: /

# Block utility URLs that should not appear in search results
Disallow: /search/
Disallow: /search?
Disallow: /*?q=
Disallow: /404/
